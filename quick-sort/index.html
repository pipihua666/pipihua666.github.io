const quickSort = (arr: number[]) => {
  if (arr.length < 2) {
    return arr // 1
  }
  const lessArr = [] // 
  const moreArr = [] // 
  const pivot = arr.shift() // ()
  arr.forEach(item => {
    if (item >= pivot) {
      moreArr.push(item)
    } else {
      lessArr.push(item)
    }
  })
  return [...quickSort(lessArr), pivot, ...quickSort(moreArr)]
}
